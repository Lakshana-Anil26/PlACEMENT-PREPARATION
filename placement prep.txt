<script>
function generateRoadmap() {
    let interest = document.getElementById("interest").value;
    let roadmapText = "";

    if (interest === "software") {
        roadmapText = `
        Year 1: Learn C/Python + Basic DSA<br>
        Year 2: Advanced DSA + Web Development<br>
        Year 3: Projects + Internships + Mock Interviews
        `;
    }
    else if (interest === "core") {
        roadmapText = `
        Year 1: Strong Basics in Core Subjects<br>
        Year 2: Mini Core Projects<br>
        Year 3: Internships + Technical Preparation
        `;
    }
    else {
        roadmapText = `
        Year 1: Strengthen Fundamentals<br>
        Year 2: Research Internships<br>
        Year 3: GATE/GRE Preparation
        `;
    }

    document.getElementById("roadmap").innerHTML = roadmapText;
}

function checkSkills() {
    let missing = [];

    if (!document.getElementById("c").checked) missing.push("C Programming");
    if (!document.getElementById("dsa").checked) missing.push("Basic DSA");
    if (!document.getElementById("html").checked) missing.push("HTML/CSS");
    if (!document.getElementById("projects").checked) missing.push("Mini Projects");

    if (missing.length === 0) {
        document.getElementById("skillsResult").innerHTML = "üéâ Great! You are on the right track!";
    } else {
        document.getElementById("skillsResult").innerHTML = "You should focus on: <br>" + missing.join("<br>");
    }
}
